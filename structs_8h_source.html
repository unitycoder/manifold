72inline HOST_DEVICE int CCW(glm::vec2 p0, glm::vec2 p1, glm::vec2 p2,
73 float tol) {
74 glm::vec2 v1 = p1 - p0;
75 glm::vec2 v2 = p2 - p0;
76 float area = v1.x * v2.y - v1.y * v2.x;
77 float base2 = glm::max(glm::dot(v1, v1), glm::dot(v2, v2));
78 if (area * area <= base2 * tol * tol)
79 return 0;
80 else
81 return area > 0 ? 1 : -1;
82}

112inline HOST_DEVICE float sind(float x) {
113 if (!std::isfinite(x)) return sin(x);
114 if (x < 0.0f) return -sind(-x);
115 int quo;
116 x = remquo(fabs(x), 90.0f, &quo);
117 switch (quo % 4) {
118 case 0:
119 return sin(glm::radians(x));
120 case 1:
121 return cos(glm::radians(x));
122 case 2:
123 return -sin(glm::radians(x));
124 case 3:
125 return -cos(glm::radians(x));
126 }
127 return 0.0f;
128}

143inline HOST_DEVICE glm::mat4x3 RotateUp(glm::vec3 up) {
144 up = glm::normalize(up);
145 glm::vec3 axis = glm::cross(up, {0, 0, 1});
146 float angle = glm::asin(glm::length(axis));
147 if (glm::dot(up, {0, 0, 1}) < 0) angle = glm::pi<float>() - angle;
148 return glm::mat4x3(glm::rotate(glm::mat4(1), angle, axis));
149}

205struct Box {
206 glm::vec3 min = glm::vec3(1.0f / 0.0f);
207 glm::vec3 max = glm::vec3(-1.0f / 0.0f);
208
212 HOST_DEVICE Box() {}
213
217 HOST_DEVICE Box(const glm::vec3 p1, const glm::vec3 p2) {
218 min = glm::min(p1, p2);
219 max = glm::max(p1, p2);
220 }
221
225 HOST_DEVICE glm::vec3 Size() const { return max - min; }
226
230 HOST_DEVICE glm::vec3 Center() const { return 0.5f * (max + min); }
231
236 HOST_DEVICE float Scale() const {
237 glm::vec3 absMax = glm::max(glm::abs(min), glm::abs(max));
238 return glm::max(absMax.x, glm::max(absMax.y, absMax.z));
239 }
240
244 HOST_DEVICE bool Contains(const Box& box) const {
245 return glm::all(glm::greaterThanEqual(box.min, min)) &&
246 glm::all(glm::greaterThanEqual(max, box.max));
247 }
248
252 HOST_DEVICE void Union(const glm::vec3 p) {
253 min = glm::min(min, p);
254 max = glm::max(max, p);
255 }
256
260 HOST_DEVICE Box Union(const Box& box) const {
261 Box out;
262 out.min = glm::min(min, box.min);
263 out.max = glm::max(max, box.max);
264 return out;
265 }
266
274 HOST_DEVICE Box Transform(const glm::mat4x3& transform) const {
275 Box out;
276 glm::vec3 minT = transform * glm::vec4(min, 1.0f);
277 glm::vec3 maxT = transform * glm::vec4(max, 1.0f);
278 out.min = glm::min(minT, maxT);
279 out.max = glm::max(minT, maxT);
280 return out;
281 }

324 HOST_DEVICE bool DoesOverlap(const Box& box) const {
325 return min.x <= box.max.x && min.y <= box.max.y && min.z <= box.max.z &&
326 max.x >= box.min.x && max.y >= box.min.y && max.z >= box.min.z;
327 }
328
333 HOST_DEVICE bool DoesOverlap(glm::vec3 p) const { // projected in z
334 return p.x <= max.x && p.x >= min.x && p.y <= max.y && p.y >= min.y;
335 }

341 return glm::all(glm::isfinite(min)) && glm::all(glm::isfinite(max));
